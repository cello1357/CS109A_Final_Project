def build_model_data(df,x_pred):
    #Predictors
    x_matrix = df[x_pred] #unpack predictor list

    # add a constant
    x_matrix = sm.add_constant(x_matrix)
    
    return x_matrix

# HOUSE - Baseline Ordinary Least Squares Regression Model 
house_train_transformed = build_model_data(house_train, x_predictors)
house_test_transformed = build_model_data(house_test, x_predictors)

#print(type(house_train_transformed))
#print(type(y_train))

house_test_transformed = house_test_transformed.astype(float)
y_train = y_train.astype(float)
y_test = y_test.astype(float)

def pca_data(X_train_df,test_df):
    learn_data=X_train_df.copy()
    learn_data.iloc[:,:]=preprocessing.scale(learn_data.iloc[:,:],axis=0)
    
    test_data=test_df.copy()
    test_data.iloc[:,:]=preprocessing.scale(test_data.iloc[:,:],axis=0)

    print ('shape of train data before pca:',X_train_df.shape)
    print ('shape of test data before pca:',test_df.shape)

    pca=PCA()
    pca.fit(learn_data)
    pca.transform(test_data)
    pca_result=pd.DataFrame({'n_components':list(np.arange(pca.n_components_)),
                             'explained_variance':list(pca.explained_variance_ratio_),
                             'cum_explained_variance':list(pca.explained_variance_ratio_.cumsum())})

    n_pca = 2
    # n_pca=pca_result.query('cum_explained_variance < 0.95').shape[0]

    # sns.scatterplot(x = 'n_components', y = 'cum_explained_variance', data = pca_result)
    print(pca_result)

    p=pca.n_components_
    cols = ['PC_{}'.format(i) for i in range(1, p+1)]
    pca_components = pd.DataFrame(pca.components_.T, columns = cols)

    learn_pca=pd.DataFrame(pca.fit_transform(learn_data),columns = cols).iloc[:,0:n_pca]
    test_pca=pd.DataFrame(pca.transform(test_data),columns = cols).iloc[:,0:n_pca]
     
    learn_pca.index = X_train_df.index
    test_pca.index = test_df.index
    print ('shape of train data after pca:',learn_pca.shape)

    return learn_pca,test_pca
